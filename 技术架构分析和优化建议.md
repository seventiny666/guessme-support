#### 响应式设计系统:
```swift
// 创建设备适配协议
protocol DeviceAdaptable {
    var isIPad: Bool { get }
    func adaptedValue<T>(_ iPadValue: T, _ iPhoneValue: T) -> T
}

extension View: DeviceAdaptable {
    var isIPad: Bool {
        UIDevice.current.userInterfaceIdiom == .pad
    }
    
    func adaptedValue<T>(_ iPadValue: T, _ iPhoneValue: T) -> T {
        isIPad ? iPadValue : iPhoneValue
    }
}

// 使用方式
.font(.system(size: adaptedValue(32, 26), weight: .bold))
```

### 3. 状态管理优化 (中优先级)

#### 创建全局状态管理:

```swift
// 应用状态管理
class AppState: ObservableObject {
    @Published var currentView: AppView = .home
    @Published var showModal: ModalType? = nil
    @Published var deviceType: DeviceType = .iPhone
    
    enum AppView {
        case home, game, purchase, customCategory
    }
    
    enum ModalType {
        case gameRules, timeSelection, gameOver
    }
}

// 在App级别注入
@main
struct GuessMeApp: App {
    @StateObject private var appState = AppState()
    
    var body: some Scene {
        WindowGroup {
            ContentView()
                .environmentObject(appState)
        }
    }
}
```
